\begin{frame}{How We Evaluated the Site: CVM}
    \begin{columns}[T]
        \begin{column}{0.49\textwidth}
            \begin{block}{What is CVM?}
                \small
                \begin{itemize}
                    \item A survey-based technique that creates a hypothetical market to elicit willingness to pay (WTP)
                    \item Respondents are asked whether they would pay a higher fee for conservation
                    \item This directly captures stated WTP for protection of monuments, gardens, and biodiversity
                \end{itemize}
            \end{block}

            \vspace{0.3em}
            \begin{block}{Why CVM?}
                \small
                \begin{itemize}
                    \item CVM captures \textbf{both} use and non-use values (Krutilla 1967; Johansson 1990)
                    \item Other methods (market pricing, hedonic, TCM) cannot capture non-use values like bequest and existence value
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.47\textwidth}
            \begin{block}{TEV Framework (Textbook)}
                \small
                Total Value = Use Value + Non-Use Value\\[4pt]
                Non-Use Value = Option Value + Bequest Value + Existence Value\\[6pt]
                \textbf{CVM is the only standard technique that captures all of these.}
            \end{block}

            \vspace{0.3em}
            \begin{block}{Non-Use Values We Captured}
                \small
                \begin{itemize}
                    \item Bequest value (preservation for future generations)
                    \item Existence value (support even without future visits)
                    \item Ecological condition perception
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Why CVM Over Other Methods?}
    \begin{table}
        \centering
        \small
        \begin{tabular}{@{}p{3.2cm}p{4.8cm}p{4.8cm}@{}}
            \toprule
            & \textbf{CVM (our choice)} & \textbf{TCM (alternative)} \\
            \midrule
            \textbf{What it measures} & Stated WTP via hypothetical market & Revealed preference via travel cost \\[3pt]
            \textbf{Use values} & \textcolor{successGreen}{\checkmark} Yes & \textcolor{successGreen}{\checkmark} Yes \\[3pt]
            \textbf{Non-use values} (bequest, existence) & \textcolor{successGreen}{\checkmark} Yes -- this is its key strength & \textcolor{alertRed}{$\times$} No -- cannot capture \\[3pt]
            \textbf{Suitability here} & Strong -- park has heritage, ecological, and intergenerational value & Weak -- 53\% visitors had multi-destination trips \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.3em}
    \keyinsight{Textbook (Ch.\ on CVM): ``Other approaches stress exclusively on estimating use values\ldots Contingent valuation [is a] general technique to elicit willingness to pay in broad and inclusive sense -- use and non-use value.''}
\end{frame}

                \item Aspects of the park visitors value

            \vspace{0.2em}
            \textbf{Phase 2 -- Valuation questions:}
            \begin{enumerate}
                \small
                \setcounter{enumi}{2}
                \item WTP for a slightly higher entry fee
                \item Maximum extra amount willing to pay
                \item Reason for refusing (protest bid identification)

\begin{frame}[shrink=5]{CVM Result 1: Willingness to Pay}
    \begin{columns}[T]
        \begin{column}{0.42\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=0.9\linewidth]{../report/figures/05_wtp_pie.png}
                \caption{Would visitors pay a higher entry fee?}
            \end{figure}
        \end{column}
        \begin{column}{0.56\textwidth}
            \begin{table}
                \centering
                \small
                \begin{tabular}{@{}lr@{}}
                    \toprule
                    \textbf{Measure} & \textbf{Value} \\
                    \midrule
                    Willing to pay more & 30 / 45 (66.7\%) \\
                    Not willing & 15 / 45 (33.3\%) \\
                    \midrule
                    Mean WTP (willing) & \rupee 90 \\
                    Median WTP (willing) & \rupee 70 \\
                    Overall mean WTP & \rupee 76.7 \\
                    \bottomrule
                \end{tabular}
            \end{table}

            \vspace{0.3em}
            \begin{block}{WTP = Current fee + Extra amount}
                \scriptsize
                Current entry fee = \rupee 50. Survey asked: ``\textit{maximum extra amount}''. By definition, WTP = total amount a person is willing to pay.\\[4pt]
                \textbf{Willing:} WTP = \rupee 50 + mean extra (\rupee 40) = \textbf{\rupee 90}\\
                \textbf{Unwilling:} WTP = \rupee 50 (they are visiting, so WTP $\geq$ \rupee 50)\\
                \textbf{Overall:} $\frac{(30 \times 90) + (15 \times 50)}{45} = \textbf{\rupee 76.7}$
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}[shrink=5]{Total Economic Value from CVM}
    \begin{columns}[T]
        \begin{column}{0.50\textwidth}
            \begin{block}{TEV Framework (Textbook)}
                \small
                Total Value = Use Value + Non-Use Value\\[3pt]
                Non-Use = Option + Bequest + Existence\\[6pt]
                \textbf{CVM captures all of these} through stated WTP (Schulze et al.\ 1981; Walsh et al.\ 1984).
            \end{block}

            \vspace{0.3em}
            \begin{table}
                \centering
                \small
                \begin{tabular}{@{}lr@{}}
                    \toprule
                    \textbf{CVM Estimate} & \textbf{Value} \\
                    \midrule
                    Overall mean WTP & \rupee 76.7/visit \\
                    \textbf{Aggregate annual WTP} & \textbf{\rupee 11.50 cr/yr} \\
                    \midrule
                    Current ticket revenue & \rupee 7.50 cr/yr \\
                    \textbf{Additional ecological value} & \textbf{\rupee 4.00 cr/yr} \\
                    \midrule
                    Bequest value (very important) & 68.9\% \\
                    Existence value (would support) & 82.2\% \\
                    \bottomrule
                \end{tabular}
            \end{table}
        \end{column}
        \begin{column}{0.46\textwidth}
            \scriptsize
            \textbf{Aggregate WTP:}\\
            $\text{\rupee}76.7 \times 15{,}00{,}000 = \textbf{\rupee 11.50 \text{ crore/year}}$\\[4pt]
            \textbf{Value already captured by market:}\\
            $\text{\rupee}50 \times 15{,}00{,}000 = \text{\rupee}7.50\text{ crore (ticket revenue)}$\\[4pt]
            \textbf{Ecological value above market price:}\\
            $\text{\rupee}11.50 - \text{\rupee}7.50 = \textbf{\rupee 4.00 \text{ crore/year}}$\\[8pt]
            \textbf{What this tells us:}\\
            The \rupee 4.00 crore is the value visitors place on Sunder Nursery's ecological and heritage conservation \textit{above} what the current ticket price captures. This includes:
            \begin{itemize}
                \item \textbf{Use value} -- enjoyment of gardens, monuments, greenery
                \item \textbf{Non-use value} -- bequest (68.9\%) and existence (82.2\%)
            \end{itemize}

            \vspace{0.2em}
            \keyinsight{Conservative estimate: bids capped at \rupee 100 extra, and only on-site visitors were surveyed.}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Discussion and Limitations}
    \begin{columns}[T]
        \begin{column}{0.52\textwidth}
            \textbf{What we can say from the CVM data:}
            \begin{itemize}
                    \item Visitors value the site well beyond the entry fee
                    \item Ecological condition is viewed positively (86.7\% Good/Excellent)
                    \item A majority (66.7\%) is willing to pay more for conservation
                    \item Strong non-use values: bequest (68.9\%) and existence (82.2\%)
            \end{itemize}

            \vspace{0.5em}
            \textbf{What this suggests:}
            \begin{itemize}
                \item A modest fee increase appears acceptable to most visitors
                \item Conservation spending is justified from both visitor preference and ecological perspectives
            \end{itemize}
        \end{column}
        \begin{column}{0.44\textwidth}
            \begin{alertblock}{CVM Limitations (from textbook)}
                \small
                \begin{itemize}
                    \item \textbf{Hypothetical bias:} WTP is stated, not actually paid
                    \item \textbf{Strategic bias:} some may under/over-state (free-rider problem)
                    \item \textbf{Information bias:} scenario description may influence responses
                    \item Sample: 45 respondents, single Friday evening
                    \item Closed-ended bids cap at \rupee 100 extra
                \end{itemize}
            \end{alertblock}

            \vspace{0.3em}
            \keyinsight{Despite these, the results are internally consistent: WTP, non-use values, and ecological perception all align.}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Conclusion}
    \begin{block}{Main message}
        \small
        Using CVM, we estimate a total WTP of \rupee 11.50 crore/year (\rupee 76.7/visitor). Since the current ticket captures only \rupee 7.50 crore, the remaining \rupee 4.00 crore/year represents ecological and heritage value not reflected in the market price.
    \end{block}

    \vspace{0.4em}
    \begin{enumerate}
        \item 66.7\% willing to pay more; mean total WTP = \rupee 90 (willing), \rupee 76.7 (overall)
        \item Strong non-use values: 68.9\% bequest, 82.2\% existence
        \item 86.7\% rated ecological condition as Good or Excellent
        \item \rupee 4.00 crore/year in ecological value above market price
    \end{enumerate}

    \vspace{0.5em}
    \begin{exampleblock}{Final takeaway}
        \centering
        \textbf{Sunder Nursery's true value is significantly higher than what its entry fee captures. CVM reveals this hidden ecological and heritage value systematically.}
    \end{exampleblock}
